#!/bin/bash
set -e

echo "ğŸ”§ Updating systemâ€¦"
sudo apt update -y
sudo apt install -y ca-certificates curl gnupg git software-properties-common openjdk-17-jdk maven

# ---------------------------------------------------------
# Install Java 21 (required for your Spring Boot backend)
# ---------------------------------------------------------
echo "ğŸ“¦ Installing Java 21..."
sudo apt install -y openjdk-21-jdk


# ---------------------------------------------------------
# Install Docker + Docker Compose
# ---------------------------------------------------------
echo "ğŸ³ Installing Docker..."

sudo rm -f /etc/apt/sources.list.d/archive_uri-https_download_docker_com_linux_ubuntu-noble.list

sudo install -m 0755 -d /etc/apt/keyrings
curl -fsSL https://download.docker.com/linux/ubuntu/gpg \
    | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg

sudo chmod a+r /etc/apt/keyrings/docker.gpg

echo "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] \
https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable" \
| sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

sudo apt update -y
sudo apt install -y docker-ce docker-ce-cli containerd.io

echo "ğŸ“¦ Installing Docker Compose v2..."
sudo curl -L \
  "https://github.com/docker/compose/releases/download/v2.23.0/docker-compose-$(uname -s)-$(uname -m)" \
  -o /usr/local/bin/docker-compose

sudo chmod +x /usr/local/bin/docker-compose


# ---------------------------------------------------------
# Install Jenkins
# ---------------------------------------------------------
echo "ğŸ›  Installing Jenkins..."

curl -fsSL https://pkg.jenkins.io/debian-stable/jenkins.io-2023.key \
    | sudo tee /usr/share/keyrings/jenkins-keyring.asc > /dev/null

echo "deb [signed-by=/usr/share/keyrings/jenkins-keyring.asc] \
https://pkg.jenkins.io/debian-stable binary/" \
| sudo tee /etc/apt/sources.list.d/jenkins.list > /dev/null

sudo apt update -y
sudo apt install -y jenkins

sudo systemctl enable jenkins
sudo systemctl start jenkins

# Jenkins needs Docker access
sudo usermod -aG docker jenkins


# ---------------------------------------------------------
# Configure Jenkins to use Java 21 (fixes mvn release error)
# ---------------------------------------------------------
echo "ğŸ”§ Configuring Jenkins JDK 21..."

sudo mkdir -p /var/lib/jenkins/tools/hudson.model.JDK/jdk21

sudo tee /var/lib/jenkins/tools/hudson.model.JDK/jdk21/config.xml > /dev/null <<EOF
<?xml version='1.1' encoding='UTF-8'?>
<jdk>
  <name>jdk21</name>
  <home>/usr/lib/jvm/java-21-openjdk-amd64</home>
  <properties/>
</jdk>
EOF

sudo chown -R jenkins:jenkins /var/lib/jenkins/tools

echo "ğŸ” Restarting Jenkins..."
sudo systemctl restart jenkins


# ---------------------------------------------------------
# Clone project repository
# ---------------------------------------------------------
echo "ğŸ“ Cloning your project repository..."
if [ ! -d "pms-leaderboard" ]; then
    git clone https://github.com/pms-org/pms-leaderboard.git
fi

echo "ğŸ‰ Setup complete!"
echo "â¡ Jenkins URL: http://<EC2-IP>:8080"
echo "â¡ Jenkins admin password: sudo cat /var/lib/jenkins/secrets/initialAdminPassword"
echo "â¡ Ensure your EC2 Security Group allows: 8080, 8081, 9092, 2181, 5432, 6379"
